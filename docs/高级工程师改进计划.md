### 2.2 错误处理重构

**问题**：异常被静默吞掉，用户不知道功能降级

**方案**：实现功能降级管理器

```python
# backend/core/degradation.py
from enum import Enum
from dataclasses import dataclass
from typing import Optional
import asyncio

class FeatureStatus(str, Enum):
    """功能状态"""
    AVAILABLE = "available"
    DEGRADED = "degraded"
    UNAVAILABLE = "unavailable"

@dataclass
class DegradationEvent:
    """降级事件"""
    feature: str
    status: FeatureStatus
    reason: str
    timestamp: float
    retry_count: int = 0

class FeatureDegradationManager:
    """功能降级管理器"""
    
    def __init__(self, ws_manager=None):
        self._features: dict[str, FeatureStatus] = {}
        self._events: list[DegradationEvent] = []
        self._ws_manager = ws_manager
        self._retry_tasks: dict[str, asyncio.Task] = {}
    
    def degrade(self, feature: str, reason: str, auto_retry: bool = True):
        """标记功能降级"""
        self._features[feature] = FeatureStatus.DEGRADED
        
        event = DegradationEvent(
            feature=feature,
            status=FeatureStatus.DEGRADED,
            reason=reason,
            timestamp=asyncio.get_event_loop().time(),
        )
        self._events.append(event)
        
        # 通知用户
        self._notify_user(feature, reason)
        
        # 记录遥测
        self._record_telemetry(event)
        
        # 自动重试
        if auto_retry:
            self._schedule_retry(feature)
    
    def is_available(self, feature: str) -> bool:
        """检查功能是否可用"""
        return self._features.get(feature, FeatureStatus.AVAILABLE) == FeatureStatus.AVAILABLE
    
    def get_status(self) -> dict[str, str]:
        """获取所有功能状态"""
        return {k: v.value for k, v in self._features.items()}
    
    def _notify_user(self, feature: str, reason: str):
        """通知用户"""
        if self._ws_manager:
            asyncio.create_task(self._ws_manager.broadcast({
                "type": "feature_degraded",
                "feature": feature,
                "reason": reason,
                "message": f"功能 {feature} 暂时不可用: {reason}",
            }))
    
    def _record_telemetry(self, event: DegradationEvent):
        """记录遥测数据"""
        # TODO: 发送到监控系统
        pass
    
    def _schedule_retry(self, feature: str):
        """安排自动重试"""
        async def retry():
            await asyncio.sleep(60)  # 1 分钟后重试
            # 尝试恢复功能
            success = await self._try_recover(feature)
            if success:
                self._features[feature] = FeatureStatus.AVAILABLE
                self._notify_user(feature, "已恢复")
        
        self._retry_tasks[feature] = asyncio.create_task(retry())
    
    async def _try_recover(self, feature: str) -> bool:
        """尝试恢复功能"""
        # 根据功能类型尝试恢复
        if feature == "vision":
            try:
                from vision.moondream import MoondreamVision
                vision = MoondreamVision()
                await vision.initialize()
                return True
            except Exception:
                return False
        return False

# 全局单例
_degradation_manager: Optional[FeatureDegradationManager] = None

def get_degradation_manager() -> FeatureDegradationManager:
    global _degradation_manager
    if _degradation_manager is None:
        _degradation_manager = FeatureDegradationManager()
    return _degradation_manager
```

```python
# backend/core/security.py
import re
from typing import Optional
from dataclasses import dataclass

@dataclass
class SecurityCheckResult:
    """安全检查结果"""
    is_safe: bool
    reason: Optional[str] = None
    risk_level: str = "low"  # low, medium, high

class SecurityGuard:
    """安全守卫"""
    
    def __init__(self):
        # Prompt 注入黑名单
        self._injection_patterns = [
            r"忽略.*指令",
            r"ignore.*instructions",
            r"你现在是",
            r"you are now",
            r"pretend.*you.*are",
            r"act.*as.*if",
            r"system.*prompt",
            r"reveal.*instructions",
        ]
        
        # 敏感信息模式
        self._sensitive_patterns = {
            "email": r'[\w.-]+@[\w.-]+\.\w+',
            "phone": r'1[3-9]\d{9}',
            "id_card": r'\d{17}[\dXx]',
            "bank_card": r'\d{16,19}',
        }
        
        # 危险命令模式
        self._dangerous_patterns = [
            r"rm\s+-rf",
            r"sudo\s+",
            r"chmod\s+777",
            r"curl.*\|\s*sh",
            r"wget.*\|\s*bash",
        ]
    
    def validate_input(self, user_input: str) -> SecurityCheckResult:
        """验证用户输入"""
        
        # 1. 检查长度
        if len(user_input) > 2000:
            return SecurityCheckResult(
                is_safe=False,
                reason="输入过长",
                risk_level="medium"
            )
        
        # 2. 检查 Prompt 注入
        for pattern in self._injection_patterns:
            if re.search(pattern, user_input, re.IGNORECASE):
                return SecurityCheckResult(
                    is_safe=False,
                    reason="检测到潜在的 Prompt 注入",
                    risk_level="high"
                )
        
        # 3. 检查危险命令
        for pattern in self._dangerous_patterns:
            if re.search(pattern, user_input, re.IGNORECASE):
                return SecurityCheckResult(
                    is_safe=False,
                    reason="检测到危险命令",
                    risk_level="high"
                )
        
        return SecurityCheckResult(is_safe=True)
    
    def sanitize_output(self, text: str) -> str:
        """净化输出，移除敏感信息"""
        
        # 替换敏感信息
        for info_type, pattern in self._sensitive_patterns.items():
            if info_type == "email":
                text = re.sub(pattern, '***@***.com', text)
            elif info_type == "phone":
                text = re.sub(pattern, '1**********', text)
            elif info_type == "id_card":
                text = re.sub(pattern, '******************', text)
            elif info_type == "bank_card":
                text = re.sub(pattern, '****', text)
        
        return text
    
    def validate_tool_call(self, tool_name: str, args: dict) -> SecurityCheckResult:
        """验证工具调用"""
        
        # 检查工具名称
        allowed_tools = [
            "search_docs", "fetch_doc_page", "highlight_element",
            "visual_locate", "screenshot_annotate", "classify_page",
            "learn_flow", "save_memory", "recall_memory",
            "browser_snapshot", "browser_interact", "browser_find",
        ]
        
        if tool_name not in allowed_tools:
            return SecurityCheckResult(
                is_safe=False,
                reason=f"未知工具: {tool_name}",
                risk_level="high"
            )
        
        # 检查参数
        if tool_name == "fetch_doc_page":
            url = args.get("url", "")
            if not url.startswith(("http://", "https://")):
                return SecurityCheckResult(
                    is_safe=False,
                    reason="无效的 URL",
                    risk_level="medium"
                )
        
        return SecurityCheckResult(is_safe=True)

# 全局单例
_security_guard: Optional[SecurityGuard] = None

def get_security_guard() -> SecurityGuard:
    global _security_guard
    if _security_guard is None:
        _security_guard = SecurityGuard()
    return _security_guard
```

### 3.1 性能优化：并行执行 + 流式返回

**问题**：串行执行导致延迟 3-9 秒

**方案**：并行执行 + 流式返回

```python
# backend/core/performance.py
import asyncio
from typing import AsyncGenerator, Any

class ParallelExecutor:
    """并行执行器"""
    
    @staticmethod
    async def execute_parallel(tasks: list[asyncio.Task]) -> list[Any]:
        """并行执行多个任务"""
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # 处理异常
        processed = []
        for i, result in enumerate(results):
            if isinstance(result, Exception):
                logger.warning(f"任务 {i} 失败: {result}")
                processed.append(None)
            else:
                processed.append(result)
        
        return processed
    
    @staticmethod
    async def execute_with_timeout(
        task: asyncio.Task,
        timeout: float,
        fallback: Any = None
    ) -> Any:
        """带超时的任务执行"""
        try:
            return await asyncio.wait_for(task, timeout=timeout)
        except asyncio.TimeoutError:
            logger.warning(f"任务超时 ({timeout}s)")
            return fallback

class StreamProcessor:
    """流式处理器"""
    
    def __init__(self):
        self._buffer = []
        self._buffer_size = 10
    
    async def process_stream(
        self,
        source: AsyncGenerator[str, None],
        transform: callable = None
    ) -> AsyncGenerator[str, None]:
        """处理流式数据"""
        async for chunk in source:
            # 转换
            if transform:
                chunk = transform(chunk)
            
            # 缓冲
            self._buffer.append(chunk)
            
            # 批量返回
            if len(self._buffer) >= self._buffer_size:
                yield "".join(self._buffer)
                self._buffer.clear()
        
        # 返回剩余数据
        if self._buffer:
            yield "".join(self._buffer)
```

### 3.2 缓存策略

**问题**：相同查询重复计算

**方案**：实现多级缓存

```python
# backend/core/cache.py
import hashlib
import json
from typing import Optional, Any
from datetime import datetime, timedelta

class CacheManager:
    """多级缓存管理器"""
    
    def __init__(self, redis_client=None):
        self._redis = redis_client
        self._local_cache: dict[str, tuple[Any, datetime]] = {}
        self._local_ttl = timedelta(minutes=5)
        self._redis_ttl = timedelta(hours=1)
    
    def _generate_key(self, prefix: str, params: dict) -> str:
        """生成缓存键"""
        # 对参数排序，确保相同参数生成相同键
        sorted_params = json.dumps(params, sort_keys=True)
        hash_value = hashlib.md5(sorted_params.encode()).hexdigest()
        return f"{prefix}:{hash_value}"
    
    async def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        
        # 1. 检查本地缓存
        if key in self._local_cache:
            value, expire_at = self._local_cache[key]
            if datetime.now() < expire_at:
                return value
            else:
                del self._local_cache[key]
        
        # 2. 检查 Redis 缓存
        if self._redis:
            try:
                data = await self._redis.get(key)
                if data:
                    value = json.loads(data)
                    # 写入本地缓存
                    self._set_local(key, value)
                    return value
            except Exception:
                pass
        
        return None
    
    async def set(self, key: str, value: Any, ttl: Optional[timedelta] = None):
        """设置缓存"""
        
        # 1. 写入本地缓存
        self._set_local(key, value)
        
        # 2. 写入 Redis 缓存
        if self._redis:
            try:
                data = json.dumps(value)
                await self._redis.setex(
                    key,
                    int((ttl or self._redis_ttl).total_seconds()),
                    data
                )
            except Exception:
                pass
    
    def _set_local(self, key: str, value: Any):
        """设置本地缓存"""
        self._local_cache[key] = (value, datetime.now() + self._local_ttl)
        
        # 清理过期缓存
        if len(self._local_cache) > 1000:
            self._cleanup_local()
    
    def _cleanup_local(self):
        """清理过期的本地缓存"""
        now = datetime.now()
        expired = [
            k for k, (_, expire_at) in self._local_cache.items()
            if now >= expire_at
        ]
        for k in expired:
            del self._local_cache[k]
    
    async def invalidate(self, pattern: str):
        """失效缓存"""
        # 清理本地缓存
        keys_to_delete = [k for k in self._local_cache if k.startswith(pattern)]
        for k in keys_to_delete:
            del self._local_cache[k]
        
        # 清理 Redis 缓存
        if self._redis:
            try:
                keys = await self._redis.keys(f"{pattern}:*")

    async def execute(self, query: str, top_k: int = 5) -> str:

        cache_key = self._cache._generate_key("search", {
            "query": query,
            "top_k": top_k,
        })
        
        # 检查缓存
        cached = await self._cache.get(cache_key)
        if cached:
            return cached
        
        # 执行搜索

        # 写入缓存
```

### 3.3 监控和可观测性

```python
# backend/core/observability.py
import time
from typing import Optional
from contextlib import contextmanager
from dataclasses import dataclass
from collections import defaultdict

@dataclass
class Metric:
    """指标"""
    name: str
    value: float
    timestamp: float
    labels: dict[str, str]

class MetricsCollector:
    """指标收集器"""
    
    def __init__(self):
        self._counters: dict[str, int] = defaultdict(int)
        self._histograms: dict[str, list[float]] = defaultdict(list)
        self._gauges: dict[str, float] = {}
    
    def increment(self, name: str, labels: dict[str, str] = None):
        """递增计数器"""
        key = self._make_key(name, labels)
        self._counters[key] += 1
    
    def observe(self, name: str, value: float, labels: dict[str, str] = None):
        """记录直方图"""
        key = self._make_key(name, labels)
        self._histograms[key].append(value)
    
    def set_gauge(self, name: str, value: float, labels: dict[str, str] = None):
        """设置仪表盘"""
        key = self._make_key(name, labels)
        self._gauges[key] = value
    
    def get_metrics(self) -> list[Metric]:
        """获取所有指标"""
        metrics = []
        timestamp = time.time()
        
        for key, value in self._counters.items():
            name, labels = self._parse_key(key)
            metrics.append(Metric(name, value, timestamp, labels))
        
        for key, values in self._histograms.items():
            name, labels = self._parse_key(key)
            if values:
                metrics.append(Metric(f"{name}_avg", sum(values) / len(values), timestamp, labels))
                metrics.append(Metric(f"{name}_p95", self._percentile(values, 95), timestamp, labels))
                metrics.append(Metric(f"{name}_p99", self._percentile(values, 99), timestamp, labels))
        
        for key, value in self._gauges.items():
            name, labels = self._parse_key(key)
            metrics.append(Metric(name, value, timestamp, labels))
        
        return metrics
    
    def _make_key(self, name: str, labels: dict[str, str] = None) -> str:
        if labels:
            label_str = ",".join(f"{k}={v}" for k, v in sorted(labels.items()))
            return f"{name}{{{label_str}}}"
        return name
    
    def _parse_key(self, key: str) -> tuple[str, dict[str, str]]:
        if "{" in key:
            name = key.split("{")[0]
            label_str = key.split("{")[1].rstrip("}")
            labels = dict(item.split("=") for item in label_str.split(","))
            return name, labels
        return key, {}
    
    def _percentile(self, values: list[float], p: int) -> float:
        sorted_values = sorted(values)
        index = int(len(sorted_values) * p / 100)
        return sorted_values[min(index, len(sorted_values) - 1)]

class Tracer:
    """追踪器"""
    
    def __init__(self):
        self._spans: list[dict] = []
    
    @contextmanager
    def start_span(self, name: str, attributes: dict = None):
        """开始一个 span"""
        span = {
            "name": name,
            "start_time": time.time(),
            "attributes": attributes or {},
        }
        
        try:
            yield span
        except Exception as e:
            span["error"] = str(e)
            raise
        finally:
            span["end_time"] = time.time()
            span["duration"] = span["end_time"] - span["start_time"]
            self._spans.append(span)
    
    def get_spans(self) -> list[dict]:
        """获取所有 spans"""
        return self._spans

# 全局实例
_metrics = MetricsCollector()
_tracer = Tracer()

def get_metrics() -> MetricsCollector:
    return _metrics

def get_tracer() -> Tracer:
    return _tracer
```

```python
# backend/core/logging.py
import structlog
import logging
from typing import Any

def setup_logging(log_level: str = "INFO"):
    """设置结构化日志"""
    
    # 配置 structlog
    structlog.configure(
        processors=[
            structlog.contextvars.merge_contextvars,
            structlog.processors.add_log_level,
            structlog.processors.StackInfoRenderer(),
            structlog.dev.set_exc_info,
            structlog.processors.TimeStamper(fmt="iso"),
            structlog.processors.JSONRenderer(),
        ],
        context_class=dict,
        logger_factory=structlog.PrintLoggerFactory(),
        wrapper_class=structlog.BoundLogger,
        cache_logger_on_first_use=True,
    )
    
    # 配置标准库日志
    logging.basicConfig(
        format="%(message)s",
        level=getattr(logging, log_level.upper()),
    )

class RequestLogger:
    """请求日志器"""
    
    def __init__(self):
        self._logger = structlog.get_logger("request")
    
    def log_request(self, request_id: str, method: str, path: str, **kwargs):
        """记录请求"""
        self._logger.info(
            "request_started",
            request_id=request_id,
            method=method,
            path=path,
            **kwargs,
        )
    
    def log_response(self, request_id: str, status_code: int, duration: float, **kwargs):
        """记录响应"""
        self._logger.info(
            "request_completed",
            request_id=request_id,
            status_code=status_code,
            duration_ms=duration * 1000,
            **kwargs,
        )
    
    def log_error(self, request_id: str, error: Exception, **kwargs):
        """记录错误"""
        self._logger.error(
            "request_failed",
            request_id=request_id,
            error_type=type(error).__name__,
            error_message=str(error),
            **kwargs,
        )

# 使用示例
request_logger = RequestLogger()

@app.middleware("http")
async def logging_middleware(request: Request, call_next):
    request_id = str(uuid.uuid4())
    start_time = time.time()
    
    # 记录请求
    request_logger.log_request(
        request_id=request_id,
        method=request.method,
        path=request.url.path,
    )
    
    try:
        response = await call_next(request)
        
        # 记录响应
        duration = time.time() - start_time
        request_logger.log_response(
            request_id=request_id,
            status_code=response.status_code,
            duration=duration,
        )
        
        return response
    except Exception as e:
        # 记录错误
        request_logger.log_error(
            request_id=request_id,
            error=e,
        )
        raise
```
