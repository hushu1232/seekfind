### T1.2 上下文窗口优化

**优先级**: P0
**工时**: 3 天
**问题**: 8GB 显存下，长上下文会占用大量 KV Cache

**方案**: 动态上下文管理

```python
# backend/memory/context_manager.py

class ContextManager:
    def __init__(self, max_tokens: int = 2048):
        self.max_tokens = max_tokens
    
    def trim_context(self, messages: list) -> list:
        """智能截断上下文"""
        total = self._count_tokens(messages)
        if total <= self.max_tokens:
            return messages
        
        # 保留系统 prompt
        system = messages[0]
        history = messages[1:]
        
        # 保留最近 5 轮对话
        recent = history[-10:]  # 5 轮 = 10 条消息
        old = history[:-10]
        
        # 摘要中间对话
        if old:
            summary = self._summarize(old)
            return [system, summary] + recent
        
        return [system] + recent
    
    def _summarize(self, messages: list) -> dict:
        """用 LLM 生成摘要"""
        prompt = "用一句话概括这段对话的主要内容：\n"
        for msg in messages:
            prompt += f"{msg['role']}: {msg['content'][:50]}\n"
        
        summary = self.llm.invoke(prompt)
        return {"role": "system", "content": f"对话摘要：{summary}"}
```

### T2.1 Reranker 精排上线

**优先级**: P0

**问题**: 当前检索直接返回 Top-K，可能不精准

**方案**: 增加 Reranker 精排层

```python
# backend/tools/search_docs.py

class HybridSearcher:
    def __init__(self):
        self.reranker = self._init_reranker()
    
    def _init_reranker(self):
        """初始化 Reranker（8GB 显存友好）"""
        try:
            from sentence_transformers import CrossEncoder
            # bge-reranker-base: 1.1GB，适合 8GB 显存
            return CrossEncoder('BAAI/bge-reranker-base')
        except Exception as e:
            logger.warning(f"Reranker 初始化失败: {e}")
            return None
    
    def search(self, query: str, top_k: int = 5) -> list:
        # 1. 向量检索 Top-20
        vector_results = self.vector_search(query, top_k=20)
        
        # 2. BM25 检索 Top-20
        bm25_results = self.bm25_search(query, top_k=20)
        
        # 3. RRF 融合
        fused = self.rrf_fusion(vector_results, bm25_results)
        
        # 4. Reranker 精排（新增）
        if self.reranker:
            reranked = self._rerank(query, fused[:20])
            return reranked[:top_k]
        
        return fused[:top_k]
    
    def _rerank(self, query: str, candidates: list) -> list:
        """Reranker 精排"""
        pairs = [(query, c['text']) for c in candidates]
        scores = self.reranker.predict(pairs)
        
        # 按分数排序
        scored = list(zip(candidates, scores))
        scored.sort(key=lambda x: x[1], reverse=True)
        
        return [item[0] for item in scored]
```

```python
# backend/tools/search_docs.py

class QueryRewriter:
    def __init__(self, llm):
        self.llm = llm
    
    async def rewrite(self, query: str) -> list:
        """将用户查询改写为多个变体"""
        prompt = f"""将用户问题改写为 2 个不同表述，每行一个，不要解释。

用户：{query}
改写："""
        
        response = await self.llm.ainvoke(prompt)
        rewrites = [line.strip() for line in response.split('\n') if line.strip()]
        
        return [query] + rewrites[:2]  # 原始 + 2 个改写

async def search_with_rewrite(self, query: str) -> list:
    """查询改写 + 多路检索"""
    
    # 1. 生成查询变体
    queries = await self.rewriter.rewrite(query)
    
    # 2. 多路检索
    all_results = []
    for q in queries:
        results = self.search(q, top_k=10)
        all_results.extend(results)
    
    # 3. 合并去重 + RRF
    merged = self.merge_and_deduplicate(all_results)
    
    # 4. Reranker 精排
    if self.reranker:
        return self._rerank(query, merged[:20])[:5]
    
    return merged[:5]
```

### T2.3 上下文感知检索

**优先级**: P1
**工时**: 2 天
**问题**: 多轮对话时，每次独立检索，丢失上下文

**方案**: 结合对话历史生成更精准的查询

```python
# backend/tools/search_docs.py

class ContextAwareSearch:
    def build_contextual_query(self, current_query: str, history: list) -> str:
        """结合历史生成查询"""
        
        if not history:
            return current_query
        
        # 提取最近 2 轮的关键信息
        recent = history[-4:]  # 2 轮 = 4 条消息
        
        # 提取关键词
        keywords = []
        for msg in recent:
            if msg['role'] == 'assistant':
                # 从助手回复中提取实体
                entities = self._extract_entities(msg['content'])
                keywords.extend(entities)
        
        # 拼接查询
        if keywords:
            context = ' '.join(keywords[-3:])  # 最多 3 个关键词
            return f"{context} {current_query}"
        
        return current_query
    
    def _extract_entities(self, text: str) -> list:
        """提取实体（简化版）"""
        # 实际实现可以用 NER 或正则
        import re
        # 提取引号内容
        quoted = re.findall(r'[""「」]([^""「」]+)[""「」]', text)
        # 提取按钮/链接等 UI 元素
        ui_elements = re.findall(r'(按钮|链接|菜单|输入框|下拉框)', text)
        return quoted + ui_elements
```

```
用户: "点哪里"         → 理解为"新建项目按钮 点哪里"
```

```typescript
// extension/content/shortcuts.ts

interface ShortcutConfig {
    key: string;
    ctrl?: boolean;
    shift?: boolean;
    alt?: boolean;
    action: string;
    description: string;
}

const DEFAULT_SHORTCUTS: ShortcutConfig[] = [
    { key: 'Q', ctrl: true, shift: true, action: 'toggleSidebar', description: '唤出/隐藏侧边栏' },
    { key: 'H', ctrl: true, shift: true, action: 'highlightLast', description: '高亮上一个目标' },
    { key: 'R', ctrl: true, shift: true, action: 'repeatLastGuide', description: '重复上一个指引' },
    { key: 'Escape', action: 'clearHighlights', description: '清除所有高亮' },
    { key: 'M', ctrl: true, shift: true, action: 'toggleMicrophone', description: '开启/关闭语音' },
];

class ShortcutManager {
    private shortcuts: Map<string, ShortcutConfig> = new Map();
    
    constructor() {
        this.loadShortcuts();
        this.bindEvents();
    }
    
    private loadShortcuts() {
        // 从 storage 加载自定义快捷键
        chrome.storage.sync.get('shortcuts', (result) => {
            const saved = result.shortcuts || DEFAULT_SHORTCUTS;
            saved.forEach(s => {
                const key = this.generateKey(s);
                this.shortcuts.set(key, s);
            });
        });
    }
    
    private generateKey(config: ShortcutConfig): string {
        const parts = [];
        if (config.ctrl) parts.push('Ctrl');
        if (config.shift) parts.push('Shift');
        if (config.alt) parts.push('Alt');
        parts.push(config.key);
        return parts.join('+');
    }
    
    private bindEvents() {
        document.addEventListener('keydown', (e) => {
            const key = this.generateKey({
                key: e.key,
                ctrl: e.ctrlKey,
                shift: e.shiftKey,
                alt: e.altKey,
            });
            
            const config = this.shortcuts.get(key);
            if (config) {
                e.preventDefault();
                e.stopPropagation();
                this.executeAction(config.action);
            }
        });
    }
    
    private executeAction(action: string) {
        chrome.runtime.sendMessage({ type: 'shortcut', action });
    }
}
```

```python
# backend/llm_manager.py

class LLMManager:
    def __init__(self, config: dict):
        self.config = config
        self.local_llm = self._init_local_llm()
        self.cloud_llm = self._init_cloud_llm()
    
    async def get_llm(self, task_type: str = 'default'):
        """根据任务类型和可用性选择 LLM"""
        
        # 1. 检查本地可用性
        if await self._check_local_available():
            return self.local_llm
        
        # 2. 检查云端配置
        if self._check_cloud_configured():
            logger.warning("本地模型不可用，降级到云端 API")
            await self._notify_user(
                "本地模型不可用，已切换云端（数据会发送到服务器）",
                level='warning'
            )
            return self.cloud_llm
        
        # 3. 都不可用
        raise LLMNotAvailableError(
            "请安装 Ollama 或配置云端 API Key\n"
            "本地模式: ollama serve && ollama pull qwen2.5:7b\n"
            "云端模式: 在设置中填入 API Key"
        )
    
    async def _check_local_available(self) -> bool:
        """检查本地 Ollama 是否可用"""
        try:
            async with httpx.AsyncClient() as client:
                response = await client.get(
                    'http://localhost:11434/api/tags',
                    timeout=2.0
                )
                return response.status_code == 200
        except Exception:
            return False
    
    def _check_cloud_configured(self) -> bool:
        """检查云端 API 是否配置"""
        return bool(self.config.get('cloud_api_key'))
    
    async def _notify_user(self, message: str, level: str = 'info'):
        """通知用户"""
        # 通过 WebSocket 发送通知
        await self.ws_manager.send({
            'type': 'notification',
            'level': level,
            'message': message,
        })
```
