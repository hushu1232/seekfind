# backend/core/security.py

class SecurityGuard:
    def __init__(self):
        # 增强注入模式
        self._injection_patterns = [
            # 中文注入
            r"忽略.*(?:之前|上面|系统).*(?:指令|提示|规则)",
            r"(?:告诉|显示|透露).*(?:系统|初始).*(?:提示|指令)",
            r"(?:你现在|你扮演|假装).*(?:是|成为|变成)",
            # 英文注入
            r"ignore\s+(?:all\s+)?(?:previous|above|system)\s+(?:instructions|prompts)",
            r"(?:you\s+are|you're)\s+now\s+(?:a|an|the)",
            # 对抗样本
            r"(?:请帮我|帮我总结).*(?:忽略|ignore).*(?:规则|rules)",
        ]

        # URL 白名单
        self._allowed_domains = {
            "github.com",
            "gitlab.com",
            "docs.google.com",
            # ...
        }

    def validate_url(self, url: str) -> bool:
        """验证 URL 是否安全"""
        from urllib.parse import urlparse
        parsed = urlparse(url)

        # 检查协议
        if parsed.scheme not in ("http", "https"):
            return False

        # 检查域名
        if parsed.hostname not in self._allowed_domains:
            return False

        # 检查内网地址
        if self._is_private_ip(parsed.hostname):
            return False

        return True

    def _is_private_ip(self, hostname: str) -> bool:
        """检查是否是内网地址"""
        import ipaddress
        try:
            ip = ipaddress.ip_address(hostname)
            return ip.is_private
        except ValueError:
            return False
```

# Step 3: 中间件记录指标
@app.middleware("http")
async def metrics_middleware(request, call_next):
    start_time = time.time()
    response = await call_next(request)
    duration = time.time() - start_time

    REQUEST_COUNT.labels(
        method=request.method,
        endpoint=request.url.path,
        status=response.status_code
    ).inc()

    REQUEST_DURATION.labels(
        method=request.method,
        endpoint=request.url.path
    ).observe(duration)

    return response

# backend/core/redis_cache.py

import aioredis

class RedisCache:
    def __init__(self, redis_url: str):
        self._redis = aioredis.from_url(redis_url)

    async def get(self, key: str):
        return await self._redis.get(key)

    async def set(self, key: str, value: str, ttl: int = 3600):
        await self._redis.setex(key, ttl, value)

    async def delete(self, key: str):
        await self._redis.delete(key)

# Step 2: 缓存预热
# backend/core/cache_warmup.py

class CacheWarmup:
    def __init__(self, cache_manager, retrieval_service):
        self._cache = cache_manager
        self._retrieval = retrieval_service

    async def warmup(self):
        """预热缓存"""
        # 获取热门查询
        hot_queries = await self._get_hot_queries()

        # 并行预热
        tasks = [
            self._warmup_query(query)
            for query in hot_queries
        ]
        await asyncio.gather(*tasks)

    async def _warmup_query(self, query: str):
        """预热单个查询"""
        cache_key = self._cache._generate_key("search", {"query": query})
        if not await self._cache.get(cache_key):
            results = await self._retrieval.search(query)
            await self._cache.set(cache_key, results)
```

from abc import ABC, abstractmethod

class VectorStore(ABC):
    @abstractmethod
    async def search(self, query_vector, top_k, filter_expr):
        pass

    @abstractmethod
    async def insert(self, vectors, metadata):
        pass

class ChromaVectorStore(VectorStore):
    """Chroma 适配器"""
    pass

class MilvusVectorStore(VectorStore):
    """Milvus 适配器"""
    pass

    """Qdrant 适配器"""
    pass

# Step 2: 工厂模式创建实例
def create_vector_store(store_type: str, config: dict) -> VectorStore:
    if store_type == "chroma":
        return ChromaVectorStore(config)
    elif store_type == "milvus":
        return MilvusVectorStore(config)

        return QdrantVectorStore(config)
